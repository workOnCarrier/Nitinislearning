bool isBalanced(const string & input){
    auto len = input.len();
    int count = 0;
    for ( auto i = 0; i < len; i++){
        if (input[i] == '(') count++;
        else if (input[i] == ')'){
            count--;
        }
        if (count < 0) return false;
    }
    if ( count != 0) return false;
    return true;
}

bool isBalancedMax(const string & input){
    std::stack<char> l_stack;
    auto len = input.len();
    int count = 0;
    int max_count =0;
    for ( auto i = 0; i < len; i++){
        if (input[i] == '('){
            l_stack.push(i);
            count++;
        }
        else if (input[i] == ')'){
            if (l_stack.empty()) { 
                max_count = max(max_count, count);
                count = 0;
            }
            else {
                l_stack.pop();
                count ++;
            }
        }
    }
    return max_count;
}
